CREATE DATABASE smallfarm;

\c smallfarm;

\\ Data Pengguna + Admin :
\\ (-)foto profil

CREATE TABLE Pengguna(
Email VARCHAR(50) NOT NULL,
Username VARCHAR(15) NOT NULL,
Password VARCHAR(50) NOT NULL,
FName VARCHAR(15) NOT NULL,
LName VARCHAR(15) NOT NULL,
BDate DATE,
Sex VARCHAR(10),
Address VARCHAR(100),
Description VARCHAR(100),
Status VARCHAR default 'user',
CONSTRAINT Pengguna_Username_PK PRIMARY KEY(Username));

\\ Artikel :
\\ (-)foto ilustrasi artikel

CREATE TABLE Artikel(
Kategori VARCHAR(50) NOT NULL,
Judul VARCHAR(150) NOT NULL,
CONSTRAINT Artikel_Judul_PK PRIMARY KEY(Judul));

\\ Penjualan :
\\ (-)foto barang

CREATE TABLE Penjualan(
Kategori VARCHAR(50) default 'Lainnya',
Id serial,
Nama_Produk VARCHAR(50) NOT NULL,
Harga_Produk float NOT NULL,
Stok int NOT NULL,
Penjual VARCHAR(15) NOT NULL,
CONSTRAINT Penjualan_Id_PK PRIMARY KEY(Id));
